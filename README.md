  function exportCSV(key) {
    let rows, headers;
    if (key === 'submissions') {
      headers = ['#','Date','Name','Email','Group Size','Budget','Trip Type','Feeling','Wish List','Status','Notes'];
      rows = getData('submissions').map((s,i)=>[i+1,s.date,s.name,s.email,s.groupSize,s.budget,s.tripType,s.feeling,s.checks,s.status,s.notes]);
    } else if (key === 'clients') {
      headers = ['#','Name','Email','Phone','Trips','Total Spent','Last Contact','VIP'];
      rows = getData('clients').map((c,i)=>[i+1,c.name,c.email,c.phone,c.trips,c.spent,c.lastContact,c.vip]);
    } else if (key === 'experiences') {
      headers = ['#','Name','Region','Price','Nights','Type','Active'];
      rows = getData('experiences').map((e,i)=>[i+1,e.name,e.region,e.price,e.nights,e.type,e.active]);
    } else if (key === 'templates') {
      headers = ['#','Name','Type','Price','Sales','Active'];
      rows = getData('templates').map((t,i)=>[i+1,t.name,t.type,t.price,t.sales,t.active]);
    } else {
      exportCSV('submissions'); exportCSV('clients'); return;
    }
    const csv = [headers, ...rows].map(r => r.map(c => `"${String(c||'').replace(/"/g,'""')}"`).join(',')).join('\n');
    const blob = new Blob([csv], {type:'text/csv'});
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = `drwf-${key}-${new Date().toLocaleDateString('en-CA')}.csv`;
    a.click();
    showToast('✦ CSV downloaded — open in Google Sheets or Excel!');
  }
